    def branch
      return if @path
      @path = File.dirname(__FILE__) + "/../../"
      svn = "svn cp -r #{last_rev} #{url} #{root}/branches/#{@branch} -m 'Creating branch for #{@branch} [#{last_rev}]'"
      puts svn
      unless @debug
        $stderr.puts `#{svn}`
        $stderr.puts `cd ../branches; svn up #{@branch}`
        $stderr.puts `svn propset svn:trunk #{last_rev} .`
      end
    end
